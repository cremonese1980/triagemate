## Objective

Implement operational safety mechanisms that protect the system during degraded conditions and shutdown:

- **Health indicators:** Custom health checks for Kafka producer connectivity and outbox backlog
- **Backlog guardrail:** Threshold-based alerting and health degradation (100 WARN / 500 DOWN)
- **Graceful shutdown:** Orderly shutdown of OutboxPublisher with in-flight completion

These mechanisms enable automated monitoring, alerting, and safe deployments.

**Step 1: Kafka Health Indicator**

**Create:** `com.triagemate.triage.health.KafkaHealthIndicator`

```java
import org.apache.kafka.clients.producer.ProducerRecord;
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;
import org.springframework.beans.factory.annotation.Qualifier;
import org.springframework.boot.actuate.health.Health;
import org.springframework.boot.actuate.health.HealthIndicator;
import org.springframework.kafka.core.KafkaTemplate;
import org.springframework.stereotype.Component;

import java.time.Instant;
import java.util.Map;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.atomic.AtomicReference;

/**
 * Health indicator for Kafka producer connectivity.
 *
 * Checks if the Kafka producer can reach bootstrap servers
 * by querying partition metadata. Does NOT send a test message.
 *
 * Tracks last successful publish timestamp from OutboxPublisher.
 */
@Component
public class KafkaHealthIndicator implements HealthIndicator {

    private static final Logger log = LoggerFactory.getLogger(KafkaHealthIndicator.class);
    private static final long METADATA_TIMEOUT_SECONDS = 5;

    private final KafkaTemplate<String, String> kafkaTemplate;
    private final AtomicReference<Instant> lastSuccessfulPublish = new AtomicReference<>();

    public KafkaHealthIndicator(
            @Qualifier("outboxKafkaTemplate")
            KafkaTemplate<String, String> kafkaTemplate
    ) {
        this.kafkaTemplate = kafkaTemplate;
    }

    @Override
    public Health health() {
        try {
            // Query cluster metadata — validates connectivity without sending a message
            kafkaTemplate.getProducerFactory()
                    .createProducer()
                    .partitionsFor("__consumer_offsets");

            Instant lastPublish = lastSuccessfulPublish.get();

            return Health.up()
                    .withDetails(Map.of(
                            "status", "Connected",
                            "lastSuccessfulPublish",
                                lastPublish != null ? lastPublish.toString() : "none"
                    ))
                    .build();

        } catch (Exception ex) {
            log.warn("Kafka health check failed: {}", ex.getMessage());

            return Health.down()
                    .withDetails(Map.of(
                            "status", "Disconnected",
                            "error", ex.getMessage() != null ? ex.getMessage() : "Unknown"
                    ))
                    .build();
        }
    }

    /**
     * Called by OutboxPublisher after each successful publish.
     * Thread-safe via AtomicReference.
     */
    public void recordSuccessfulPublish() {
        lastSuccessfulPublish.set(Instant.now());
    }
}
```

**Why `partitionsFor()` instead of sending a test message:**
- Read-only operation — no side effects
- Validates full producer → bootstrap → broker connectivity
- Fast (timeout configurable)
- Standard pattern for Kafka health checks

**Why `AtomicReference<Instant>` for last publish:**
- OutboxPublisher calls `recordSuccessfulPublish()` from the scheduler thread
- Health endpoint reads from HTTP thread
- `AtomicReference` provides safe cross-thread visibility without locking

**Step 2: Outbox Backlog Health Indicator**

**Create:** `com.triagemate.triage.health.OutboxBacklogHealthIndicator`

```java
import com.triagemate.triage.persistence.JdbcOutboxRepository;
import org.springframework.boot.actuate.health.Health;
import org.springframework.boot.actuate.health.HealthIndicator;
import org.springframework.boot.actuate.health.Status;
import org.springframework.stereotype.Component;

import java.util.Map;

/**
 * Health indicator for outbox backlog.
 *
 * Thresholds:
 *   UP:        pending <= 100
 *   DEGRADED:  100 < pending <= 500
 *   DOWN:      pending > 500
 *
 * Rationale: 100 msgs ~ 1 minute of publishing at normal rate;
 * 500 msgs indicates severe degradation.
 */
@Component
public class OutboxBacklogHealthIndicator implements HealthIndicator {

    /** Custom status for degraded but not critical state. */
    static final Status DEGRADED = new Status("DEGRADED");

    static final long WARN_THRESHOLD = 100;
    static final long CRITICAL_THRESHOLD = 500;

    private final JdbcOutboxRepository repository;

    public OutboxBacklogHealthIndicator(JdbcOutboxRepository repository) {
        this.repository = repository;
    }

    @Override
    public Health health() {
        long pendingCount = repository.countPending();
        double oldestAge = repository.oldestPendingAgeSeconds();

        Map<String, Object> details = Map.of(
                "pendingCount", pendingCount,
                "oldestMessageAgeSeconds", oldestAge,
                "warnThreshold", WARN_THRESHOLD,
                "criticalThreshold", CRITICAL_THRESHOLD
        );

        if (pendingCount > CRITICAL_THRESHOLD) {
            return Health.status(new Status("DOWN"))
                    .withDetails(details)
                    .build();
        }

        if (pendingCount > WARN_THRESHOLD) {
            return Health.status(DEGRADED)
                    .withDetails(details)
                    .build();
        }

        return Health.up()
                .withDetails(details)
                .build();
    }
}
```

**Why custom `DEGRADED` status:**
- Spring Boot only provides `UP`, `DOWN`, `OUT_OF_SERVICE`, `UNKNOWN`
- `DEGRADED` signals "working but under pressure" — different from `DOWN`
- Requires `StatusAggregator` configuration to include in health rollup

```java
import org.slf4j.Logger;
import org.slf4j.LoggerFactory;

@Component
public class OutboxBacklogHealthIndicator implements HealthIndicator {

    private static final Logger log = LoggerFactory.getLogger(OutboxBacklogHealthIndicator.class);

    static final Status DEGRADED = new Status("DEGRADED");
    static final long WARN_THRESHOLD = 100;
    static final long CRITICAL_THRESHOLD = 500;

    private final JdbcOutboxRepository repository;

    public OutboxBacklogHealthIndicator(JdbcOutboxRepository repository) {
        this.repository = repository;
    }

    @Override
    public Health health() {
        long pendingCount = repository.countPending();
        double oldestAge = repository.oldestPendingAgeSeconds();

        Map<String, Object> details = Map.of(
                "pendingCount", pendingCount,
                "oldestMessageAgeSeconds", oldestAge,
                "warnThreshold", WARN_THRESHOLD,
                "criticalThreshold", CRITICAL_THRESHOLD
        );

        if (pendingCount > CRITICAL_THRESHOLD) {
            log.error(
                    "Outbox backlog CRITICAL: pending={} oldest_age_seconds={} threshold={}",
                    pendingCount, oldestAge, CRITICAL_THRESHOLD
            );
            return Health.status(new Status("DOWN"))
                    .withDetails(details)
                    .build();
        }

        if (pendingCount > WARN_THRESHOLD) {
            log.warn(
                    "Outbox backlog elevated: pending={} oldest_age_seconds={} threshold={}",
                    pendingCount, oldestAge, WARN_THRESHOLD
            );
            return Health.status(DEGRADED)
                    .withDetails(details)
                    .build();
        }

        return Health.up()
                .withDetails(details)
                .build();
    }
}
```

**Log level rationale:**
- `WARN` at 100: operational awareness — investigate if persistent
- `ERROR` at 500: severe degradation — immediate attention required
- No log at healthy state — avoid noisy health check logs

**Note on logging frequency:**
- Health check runs on each Prometheus scrape (default: 15s)
- WARN/ERROR will repeat every 15s while backlog remains elevated
- This is intentional — persistent alerting until resolved
- If log volume is a concern, add rate-limiting (e.g., log once per minute) in V2

### 11.5.c — Graceful Shutdown

#### Goal

Ensure `OutboxPublisher` stops cleanly during application shutdown:
- Stop polling for new batches
- Allow in-flight publish attempts to complete or timeout
- Release any `FOR UPDATE SKIP LOCKED` locks
- Health endpoint returns 503 after shutdown initiated

---

#### Technical Implementation

**Problem:**
- `OutboxPublisher` uses `@Scheduled(fixedDelay)` — Spring will cancel the task on shutdown
- If cancelled mid-batch, in-flight `kafkaTemplate.send().get()` may be interrupted
- `FOR UPDATE SKIP LOCKED` rows remain locked until `locked_until` expires (30s default)
- No signal to health endpoint that shutdown is in progress

**Solution:**
- Implement `SmartLifecycle` to control shutdown ordering
- Use `AtomicBoolean running` flag to stop accepting new batches
- Wait for in-flight batch to complete before signalling stop
- Health endpoint checks `running` flag

---

**Step 1: Refactor OutboxPublisher with SmartLifecycle**

**Update:** `com.triagemate.triage.outbox.OutboxPublisher`

```java
import org.springframework.context.SmartLifecycle;

import java.util.concurrent.CountDownLatch;
import java.util.concurrent.TimeUnit;
import java.util.concurrent.atomic.AtomicBoolean;

@Component
public class OutboxPublisher implements SmartLifecycle {

    private static final Logger log = LoggerFactory.getLogger(OutboxPublisher.class);

    private final JdbcOutboxRepository repository;
    private final KafkaTemplate<String, String> kafkaTemplate;
    private final OutboxProperties properties;
    private final OutboxMetrics metrics;
    private final KafkaHealthIndicator kafkaHealth;
    private final String instanceId;

    /** Lifecycle state: true while publisher is active. */
    private final AtomicBoolean running = new AtomicBoolean(false);

    /** Signalled when an in-flight batch completes. */
    private final CountDownLatch batchComplete = new CountDownLatch(1);

    /** Tracks whether a batch is currently being processed. */
    private final AtomicBoolean batchInFlight = new AtomicBoolean(false);

    public OutboxPublisher(
            JdbcOutboxRepository repository,
            @Qualifier("outboxKafkaTemplate")
            KafkaTemplate<String, String> kafkaTemplate,
            OutboxProperties properties,
            OutboxMetrics metrics,
            KafkaHealthIndicator kafkaHealth
    ) {
        this.repository = repository;
        this.kafkaTemplate = kafkaTemplate;
        this.properties = properties;
        this.metrics = metrics;
        this.kafkaHealth = kafkaHealth;
        this.instanceId = resolveHostname() + "-" + UUID.randomUUID();
    }

    private static String resolveHostname() {
        try {
            return InetAddress.getLocalHost().getHostName();
        } catch (UnknownHostException e) {
            return "unknown";
        }
    }

    // ─── SmartLifecycle ─────────────────────────────────────

    @Override
    public void start() {
        running.set(true);
        log.info("OutboxPublisher started, instanceId={}", instanceId);
    }

    @Override
    public void stop() {
        stop(() -> {});
    }

    /**
     * Graceful stop with callback.
     *
     * Signals the publisher to stop accepting new batches,
     * then waits for any in-flight batch to complete.
     */
    @Override
    public void stop(Runnable callback) {
        log.info("OutboxPublisher shutting down, waiting for in-flight batch...");
        running.set(false);

        if (batchInFlight.get()) {
            try {
                // Wait for current batch to finish (max: lock duration)
                boolean completed = batchComplete.await(
                        properties.getLockDurationSeconds(), TimeUnit.SECONDS
                );
                if (!completed) {
                    log.warn("OutboxPublisher shutdown timed out waiting for in-flight batch");
                }
            } catch (InterruptedException e) {
                Thread.currentThread().interrupt();
                log.warn("OutboxPublisher shutdown interrupted");
            }
        }

        log.info("OutboxPublisher stopped");
        callback.run();
    }

    @Override
    public boolean isRunning() {
        return running.get();
    }

    /**
     * Phase determines shutdown order.
     * Lower phase = stopped later.
     * OutboxPublisher should stop BEFORE Kafka producer (which is at phase 0).
     */
    @Override
    public int getPhase() {
        return Integer.MAX_VALUE - 100;
    }

    // ─── Scheduled polling ──────────────────────────────────

    @Scheduled(fixedDelayString = "${triagemate.outbox.poll-interval-ms:2000}",
            initialDelayString = "${triagemate.outbox.initial-delay-ms:2000}")
    public void poll() {

        if (!running.get()) {
            return;  // Shutdown in progress — skip
        }

        List<OutboxEvent> batch =
                repository.claimBatch(properties.getBatchSize(), instanceId);

        if (batch.isEmpty()) {
            return;
        }

        batchInFlight.set(true);
        try {
            for (OutboxEvent event : batch) {
                publish(event);
            }
        } finally {
            batchInFlight.set(false);
            batchComplete.countDown();
        }
    }

    private void publish(OutboxEvent event) {

        try {
            kafkaTemplate.send(
                    event.getAggregateType(),
                    event.getAggregateId(),
                    event.getPayload()
            ).get();

            repository.markPublished(event.getId());
            metrics.recordPublished();
            kafkaHealth.recordSuccessfulPublish();

        } catch (Exception ex) {

            metrics.recordKafkaFailure();

            int attempts = event.getPublishAttempts() + 1;

            if (attempts >= properties.getMaxAttempts()) {

                log.error("Outbox exhausted for event {}", event.getId(), ex);
                repository.markExhausted(event.getId(), attempts, ex.getMessage());
                return;
            }

            metrics.recordRetry();

            long delay = computeBackoff(attempts);
            Instant nextAttempt = Instant.now().plusMillis(delay);

            log.warn(
                    "Outbox publish failed (attempt {}), retry in {} ms for event {}",
                    attempts, delay, event.getId()
            );

            repository.markFailed(
                    event.getId(),
                    attempts,
                    nextAttempt,
                    ex.getMessage()
            );
        }
    }

    private long computeBackoff(int attempts) {
        long base = properties.getBaseBackoffMillis();
        long exp = base * (1L << (attempts - 1));
        return Math.min(exp, properties.getMaxBackoffMillis());
    }
}
```

**Key design decisions:**

| Decision | Rationale |
|----------|-----------|
| `AtomicBoolean running` | Thread-safe flag checked at poll entry |
| `CountDownLatch batchComplete` | Blocks shutdown until in-flight batch finishes |
| `AtomicBoolean batchInFlight` | Prevents shutdown from waiting when no batch is active |
| `getPhase() = MAX - 100` | Ensures OutboxPublisher stops before Kafka producer teardown |
| `batchComplete.await(lockDuration)` | Timeout matches lock expiry — no indefinite hang |

**Lock release on shutdown:**
- If `poll()` completes normally, all claimed rows are either `PUBLISHED` or `FAILED`
- If shutdown interrupts before `poll()` starts, no rows are claimed
- If the process crashes mid-batch, `locked_until` expiry (30s) releases locks automatically
- No manual lock cleanup needed

```java
    void poll_shouldSkipWhenNotRunning() {

        JdbcOutboxRepository repository = mock(JdbcOutboxRepository.class);
        OutboxPublisher publisher = createPublisher(repository);
        // NOT started — running = false

        // When
        publisher.poll();

        // Then — claimBatch never called

    @Test
    void stop_shouldWaitForInFlightBatch() throws Exception {
        // Given
        JdbcOutboxRepository repository = mock(JdbcOutboxRepository.class);
        OutboxPublisher publisher = createPublisher(repository);
        publisher.start();

        // Simulate slow batch processing
        CountDownLatch publishStarted = new CountDownLatch(1);
        when(repository.claimBatch(anyInt(), anyString()))
                .thenAnswer(inv -> {
                    publishStarted.countDown();
                    Thread.sleep(500);  // Simulate work
                    return Collections.emptyList();
                });

        // Start polling in background
        Thread pollThread = new Thread(publisher::poll);
        pollThread.start();
        publishStarted.await(2, TimeUnit.SECONDS);

        // When — initiate shutdown
        long startTime = System.currentTimeMillis();
        publisher.stop();
        long elapsed = System.currentTimeMillis() - startTime;

        // Then — shutdown waited for batch
        assertThat(publisher.isRunning()).isFalse();
        pollThread.join(2000);
    }

    @Test
    void stop_shouldReturnImmediatelyWhenNoBatchInFlight() {
        // Given
        JdbcOutboxRepository repository = mock(JdbcOutboxRepository.class);
        OutboxPublisher publisher = createPublisher(repository);
        publisher.start();

        // When
        long startTime = System.currentTimeMillis();
        publisher.stop();
        long elapsed = System.currentTimeMillis() - startTime;

        // Then — no waiting
        assertThat(elapsed).isLessThan(1000);
        assertThat(publisher.isRunning()).isFalse();
    }

    @Test
    void isRunning_shouldReflectLifecycleState() {
        OutboxPublisher publisher = createPublisher(mock(JdbcOutboxRepository.class));

        assertThat(publisher.isRunning()).isFalse();

        publisher.start();
        assertThat(publisher.isRunning()).isTrue();

        publisher.stop();
        assertThat(publisher.isRunning()).isFalse();
    }

        OutboxProperties properties = new OutboxProperties();
        OutboxMetrics metrics = new OutboxMetrics(

        );
```
